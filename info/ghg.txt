import React, { useState } from 'react';
import axios from 'axios';
import { YouTubeEmbed, FacebookEmbed, InstagramEmbed } from 'react-social-media-embed';

const App = () => {
  const [url, setUrl] = useState('');
  const [preview, setPreview] = useState(null);
  const [error, setError] = useState('');

  const handleDownload = async () => {
    try {
      const response = await axios.post('http://localhost:5000/download', { url }, { responseType: 'blob' });
      const urlBlob = window.URL.createObjectURL(new Blob([response.data]));
      const link = document.createElement('a');
      link.href = urlBlob;
      link.setAttribute('download', 'media.mp4');
      document.body.appendChild(link);
      link.click();
      link.remove();
    } catch (error) {
      setError('Error downloading the media');
      console.error('Error downloading the media', error);
    }
  };

  const handlePreview = () => {
    setPreview(url);
  };

  const renderPreview = () => {
    if (url.includes('youtube.com')) {
      return <YouTubeEmbed url={preview} />;
    } else if (url.includes('facebook.com')) {
      return <FacebookEmbed url={preview} />;
    } else if (url.includes('instagram.com')) {
      return <InstagramEmbed url={preview} />;
    } else {
      return <p>Unsupported URL</p>;
    }
  };

  return (
    <div className="min-h-screen bg-gray-100 flex flex-col items-center justify-center">
      <div className="bg-white p-6 rounded shadow-md w-full max-w-md">
        <input
          type="text"
          value={url}
          onChange={(e) => setUrl(e.target.value)}
          placeholder="Enter social media link"
          className="w-full p-2 border border-gray-300 rounded mb-4"
        />
        <button
          onClick={handlePreview}
          className="w-full bg-blue-500 text-white p-2 rounded mb-4"
        >
          Preview
        </button>
        {preview && renderPreview()}
        <button
          onClick={handleDownload}
          className="w-full bg-green-500 text-white p-2 rounded"
        >
          Download
        </button>
        {error && <p className="text-red-500 mt-4">{error}</p>}
      </div>
    </div>
  );
};

export default App;






const express = require('express');
const axios = require('axios');
const ytdl = require('ytdl-core');
const cors = require('cors');
const app = express();
const PORT = process.env.PORT || 5000;

app.use(cors());
app.use(express.json());

const downloadYouTube = (url, res) => {
  res.header('Content-Disposition', 'attachment; filename="video.mp4"');
  ytdl(url, { format: 'mp4' }).pipe(res);
};

const downloadFacebook = async (url, res) => {
  try {
    const response = await axios.get(url, { responseType: 'stream' });
    res.header('Content-Disposition', 'attachment; filename="video.mp4"');
    response.data.pipe(res);
  } catch (error) {
    res.status(500).send('Error downloading Facebook video');
  }
};

const downloadInstagram = async (url, res) => {
  try {
    const response = await axios.get(url, { responseType: 'stream' });
    res.header('Content-Disposition', 'attachment; filename="video.mp4"');
    response.data.pipe(res);
  } catch (error) {
    res.status(500).send('Error downloading Instagram video');
  }
};

app.post('/download', async (req, res) => {
  const { url } = req.body;
  if (ytdl.validateURL(url)) {
    downloadYouTube(url, res);
  } else if (url.includes('facebook.com')) {
    await downloadFacebook(url, res);
  } else if (url.includes('instagram.com')) {
    await downloadInstagram(url, res);
  } else {
    res.status(400).send('Unsupported URL');
  }
});

app.listen(PORT, () => {
  console.log(`Server running on port ${PORT}`);
});
