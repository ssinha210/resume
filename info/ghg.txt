// server.js
const express = require('express');
const axios = require('axios');
const ytdl = require('ytdl-core');
const cors = require('cors');

const app = express();
const PORT = process.env.PORT || 5000;

app.use(cors({ origin: 'http://localhost:3000' })); // Ensure CORS is configured for your client
app.use(express.json());

const downloadYouTube = (url, res) => {
  res.header('Content-Disposition', 'attachment; filename="video.mp4"');
  ytdl(url, { format: 'mp4' }).pipe(res);
};

const downloadFacebook = async (url, res) => {
  try {
    const response = await axios.get(url, { responseType: 'stream' });
    res.header('Content-Disposition', 'attachment; filename="video.mp4"');
    response.data.pipe(res);
  } catch (error) {
    res.status(500).send('Error downloading Facebook video');
  }
};

const downloadInstagram = async (url, res) => {
  try {
    const response = await axios.get(url, { responseType: 'stream' });
    res.header('Content-Disposition', 'attachment; filename="video.mp4"');
    response.data.pipe(res);
  } catch (error) {
    res.status(500).send('Error downloading Instagram video');
  }
};

app.post('/download', async (req, res) => {
  const { url } = req.body;

  if (ytdl.validateURL(url)) {
    downloadYouTube(url, res);
  } else if (url.includes('facebook.com')) {
    await downloadFacebook(url, res);
  } else if (url.includes('instagram.com')) {
    await downloadInstagram(url, res);
  } else {
    res.status(400).send('Unsupported URL');
  }
});

app.listen(PORT, () => {
  console.log(`Server running on port ${PORT}`);
});
